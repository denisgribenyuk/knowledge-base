## Бинарный поиск _O(log n)_
```ts
// Бинарный поиск работает только с отсортироваными списками.
// Алгоритм его работы заключается в просмотре элемента в середине диапазона с последующим исключением из поиска оставшихся элементов слева или справа.
function binarySearch(sortedList, searchItem) {
	// Создаем два указателя на начало и конец диапазона.
	let min = 0;
	let max = sortedList.length - 1;
		// Запускаем цикл в котором:
		while (min <= max) {
		// Получаем индекс середины диапазона.
		let mid = Math.floor((min + max) / 2);
		// Получаем элемент по индексу из середины диапазона.
		let guess = sortedList[mid];
		// Проверяем, искомый элемент найден! Выходим из цикла, возвращаем его индекс в списке.
		if (guess === searchItem) { return mid; };
		// Если элемент из середины диапазона больше искомого элемента,
		// то отсекаем всю правую часть диапазона так как нашего элемента там точно нет.
		// Если меньше - отсекаем левую часть.
		// Начинаем поиск заново, но уже с урезанным диапазоном.
		guess > searchItem ? max = mid - 1 : min = mid + 1;
		}
	
	return null;
};
```
